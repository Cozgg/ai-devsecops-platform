```mermaid

C4Component
  title C4 Level 3 - Component Diagram for Backend and Worker

      Component(authModule, "Authentication Module", "Django Auth / DRF", "Xác thực người dùng")

      Component(scanModule, "Scan Module", "DRF API", "Tạo scan job, upload source code và theo dõi trạng thái")
      Component(resultModule, "Result Module", "DRF API", "Cung cấp findings và AI report cho frontend")
      Component(knowledgeModule, "Knowledge Base Module", "DRF API", "Quản lý tài liệu phục vụ RAG")
  }

  Container_Boundary(worker, "Celery Worker") {
      Component(scanProcessor, "Scan Processor", "Celery Task", "Điều phối quá trình giải nén, detect ngôn ngữ và chạy scanner")
      Component(scannerService, "Scanner Service", "Semgrep / Trivy Runner", "Chạy scanner và nhận kết quả JSON")
      Component(findingService, "Finding Service", "Python Service", "Chuẩn hóa findings và tính mức độ rủi ro")
      Component(aiService, "AI Report Service", "LangChain + RAG", "Truy xuất knowledge base và sinh báo cáo AI")
  }

  ContainerDb(database, "PostgreSQL", "Database", "Lưu user, project, scan job, findings, AI report và knowledge base")
  ContainerDb(redis, "Redis", "Message Broker", "Queue trung gian cho scan job")
  Container(storage, "File Storage", "Local / Media Storage", "Lưu source code upload và source đã giải nén")
  System_Ext(scannerTools, "Scanner Tools", "Semgrep / Trivy / npm audit")
  System_Ext(llmApi, "AI Model API", "Mô hình AI dùng để sinh báo cáo")

  Rel(authModule, database, "Đọc/ghi user")
  Rel(projectModule, database, "Đọc/ghi project")

  Rel(scanModule, storage, "Lưu source upload")
  Rel(scanModule, database, "Tạo scan job")
  Rel(scanModule, redis, "Đẩy scan job vào queue")

  Rel(resultModule, database, "Đọc findings và AI report")
  Rel(knowledgeModule, database, "Đọc/ghi knowledge base")

  Rel(scanProcessor, redis, "Nhận scan job")
  Rel(scanProcessor, storage, "Đọc source code")
  Rel(scanProcessor, scannerService, "Yêu cầu scan source")

  Rel(scannerService, scannerTools, "Chạy scanner")
  Rel(scannerService, findingService, "Gửi scanner result")

  Rel(findingService, database, "Lưu findings và risk metadata")

  Rel(aiService, database, "Đọc findings và knowledge base")
  Rel(aiService, llmApi, "Gọi AI model")
  Rel(aiService, database, "Lưu AI report")
```
